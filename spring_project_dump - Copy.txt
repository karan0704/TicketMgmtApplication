    <script>
        // API Configuration
        const API_BASE_URL = 'http://localhost:8080/api';
        
        // Validation utilities
        const ValidationUtils = {
            validateEmail: (email) => {
                const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
                return re.test(email);
            },
            
            validatePhone: (phone) => {
                const re = /^[\+]?[\d\s\-\(\)]{10,}$/;
                return re.test(phone);
            },
            
            validateRequired: (value) => {
                return value && value.trim().length > 0;
            },
            
            validateMinLength: (value, min) => {
                return value && value.trim().length >= min;
            },
            
            validateMaxLength: (value, max) => {
                return value && value.trim().length <= max;
            }
        };

        // Form validation handler
        const FormValidator = {
            validateCustomerForm: (formData) => {
                const errors = {};
                
                if (!ValidationUtils.validateRequired(formData.name)) {
                    errors.name = 'Name is required';
                } else if (!ValidationUtils.validateMinLength(formData.name, 2)) {
                    errors.name = 'Name must be at least 2 characters';
                }
                
                if (!ValidationUtils.validateRequired(formData.email)) {
                    errors.email = 'Email is required';
                } else if (!ValidationUtils.validateEmail(formData.email)) {
                    errors.email = 'Please enter a valid email address';
                }
                
                if (formData.phoneNumber && !ValidationUtils.validatePhone(formData.phoneNumber)) {
                    errors.phoneNumber = 'Please enter a valid phone number';
                }
                
                if (!ValidationUtils.validateRequired(formData.companyName)) {
                    errors.companyName = 'Company name is required';
                }
                
                return errors;
            },
            
            validateEngineerForm: (formData) => {
                const errors = {};
                
                if (!ValidationUtils.validateRequired(formData.name)) {
                    errors.name = 'Name is required';
                } else if (!ValidationUtils.validate
